### Step 2: Create DeleteAccountDialog Component
```typescript
// src/domains/user/profile/5-account/ui/components/DeleteAccountDialog.tsx
import { useState } from 'react'
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogFooter,
  DialogHeader,
  DialogTitle
} from '@/components/ui/dialog'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { Label } from '@/components/ui/label'
import { AlertCircle } from 'lucide-react'
import { DeletionWarning } from './DeletionWarning'

interface DeleteAccountDialogProps {
  open: boolean
  onClose: () => void
  onConfirm: () => Promise<void>
}

export function DeleteAccountDialog({ open, onClose, onConfirm }: DeleteAccountDialogProps) {
  const [step, setStep] = useState<1 | 2>(1)
  const [confirmation, setConfirmation] = useState('')
  const [loading, setLoading] = useState(false)

  const requiredText = 'DELETE'

  const handleContinue = () => {
    if (step === 1) {
      setStep(2)
    } else {
      handleConfirm()
    }
  }

  const handleConfirm = async () => {
    if (confirmation !== requiredText) return

    setLoading(true)
    try {
      await onConfirm()
      onClose()
      setStep(1)
      setConfirmation('')
    } catch (error) {
      console.error('Failed to schedule deletion:', error)
    } finally {
      setLoading(false)
    }
  }

  const handleClose = () => {
    setStep(1)
    setConfirmation('')
    onClose()
  }

  return (
    <Dialog open={open} onOpenChange={handleClose}>
      <DialogContent className="max-w-2xl">
        <DialogHeader>
          <DialogTitle className="text-xl font-bold">
            {step === 1 ? 'Delete Account?' : 'Final Confirmation'}
          </DialogTitle>
          <DialogDescription>
            {step === 1
              ? 'You are about to delete your account. This action has serious consequences.'
              : 'Please type DELETE to confirm you understand the consequences.'}
          </DialogDescription>
        </DialogHeader>

        {step === 1 ? (
          <div className="space-y-4">
            <DeletionWarning />

            <div className="bg-muted p-4 rounded-lg">
              <p className="text-sm font-medium mb-2">What happens next:</p>
              <ol className="text-sm space-y-1 list-decimal list-inside">
                <li>Your account will be scheduled for deletion</li>
                <li>You have 24 hours to cancel this request</li>
                <li>After 24 hours, your account is permanently deleted</li>
                <li>You will receive a confirmation email</li>
              </ol>
            </div>
          </div>
        ) : (
          <div className="space-y-4">
            <DeletionWarning />

            <div className="space-y-2">
              <Label htmlFor="confirmation">Type "{requiredText}" to confirm</Label>
              <Input
                id="confirmation"
                value={confirmation}
                onChange={(e) => setConfirmation(e.target.value)}
                placeholder={requiredText}
                className="uppercase"
              />
              {confirmation && confirmation !== requiredText && (
                <p className="text-sm text-destructive flex items-center gap-1">
                  <AlertCircle className="w-4 h-4" />
                  Please type DELETE exactly as shown
                </p>
              )}
            </div>
          </div>
        )}

        <DialogFooter className="gap-2">
          <Button
            variant="outline"
            onClick={handleClose}
            disabled={loading}
          >
            Cancel
          </Button>
          <Button
            variant="destructive"
            onClick={handleContinue}
            disabled={step === 2 && confirmation !== requiredText || loading}
          >
            {loading ? 'Processing...' : step === 1 ? 'Continue' : 'Delete Account'}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  )
}
```

### Step 3: Create Main AccountManagement Component
```typescript
// src/domains/user/profile/5-account/ui/components/AccountManagement.tsx
import { useState } from 'react'
import { Card } from '@/components/ui/card'
import { Button } from '@/components/ui/button'
import { Alert, AlertDescription } from '@/components/ui/alert'
import { Calendar, X, CheckCircle } from 'lucide-react'
import { DeleteAccountDialog } from './DeleteAccountDialog'
import { useScheduleDeletion, useCancelDeletion } from '../../_shared/hooks'
import { formatDistanceToNow } from 'date-fns'

interface AccountManagementProps {
  userId: string
  deletionScheduled: string | null
}

export function AccountManagement({ userId, deletionScheduled }: AccountManagementProps) {
  const [dialogOpen, setDialogOpen] = useState(false)
  const scheduleDeletion = useScheduleDeletion()
  const cancelDeletion = useCancelDeletion()

  const handleScheduleDeletion = async () => {
    await scheduleDeletion.mutateAsync(userId)
  }

  const handleCancelDeletion = async () => {
    await cancelDeletion.mutateAsync(userId)
  }

  const isDeletionScheduled = deletionScheduled !== null
  const deletionDate = deletionScheduled ? new Date(deletionScheduled) : null

  return (
    <div className="space-y-6">
      <Card className="p-6">
        <h2 className="text-lg font-semibold mb-4">Account Management</h2>
        <p className="text-sm text-muted-foreground mb-6">
          Manage your account settings and deletion
        </p>

        {isDeletionScheduled && deletionDate ? (
          <Alert variant="destructive" className="mb-6">
            <Calendar className="h-4 w-4" />
            <AlertDescription className="flex items-center justify-between">
              <span>
                Account deletion scheduled for{' '}
                {formatDistanceToNow(deletionDate, { addSuffix: true })}
              </span>
              <Button
                size="sm"
                variant="outline"
                onClick={handleCancelDeletion}
                disabled={cancelDeletion.isPending}
              >
                <X className="w-4 h-4 mr-1" />
                Cancel
              </Button>
            </AlertDescription>
          </Alert>
        ) : (
          <Alert>
            <CheckCircle className="h-4 w-4" />
            <AlertDescription>
              Your account is active and not scheduled for deletion.
            </AlertDescription>
          </Alert>
        )}
      </Card>

      <Card className="p-6 border-destructive/50">
        <h3 className="text-lg font-semibold mb-2 text-destructive">Danger Zone</h3>
        <p className="text-sm text-muted-foreground mb-4">
          Once you delete your account, there is no going back. Please be certain.
        </p>

        <Button
          variant="destructive"
          onClick={() => setDialogOpen(true)}
          disabled={isDeletionScheduled}
        >
          Delete Account
        </Button>
      </Card>

      <DeleteAccountDialog
        open={dialogOpen}
        onClose={() => setDialogOpen(false)}
        onConfirm={handleScheduleDeletion}
      />
    </div>
  )
}
```
